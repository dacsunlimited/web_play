# AdCreative
    {
      version: Integer,
      title: String,
      type: String ['text', 'image'],
      creative: {
        text: String, 
        image: String # only valid when type is image
      },
      link: String
    }

# sample
    {
      "version": 1,
      "title": 'default add 1',
      "type": 'text',
      "creative": {
        "text": "Work Hard, PLAY Hard", 
        "image": ""
      },
      "link": 'http://playfoundation.dev'
    }

# PositionPrice
    {
      id: String,
      duration: String, #['1d', '1w', '1m']
      price: Integer, #10000000
      asset: String, # PLS
    }

# sample
    {
      "id": "plain1w",
      "duration": "1w",
      "price": 100000000,
      "asset": "PLS"
    }
